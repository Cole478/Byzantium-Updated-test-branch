# BYZANTINE RENAMES

## The goal here is to add renames for every state the Byz takes.  Of course there are 700+ states so the chances of me ever getting to them all is pretty slim.

add_namespace = byznames

news_event = {
	id = byznames.0
	hidden = yes
	
	fire_only_once = yes
	
	trigger = {
		GRE = { owns_state = 182 }
	}

	immediate = {
		182 = { set_state_name = "Krete" }
	}
	
	option = {
		name = byznames.0.a
	}
}
