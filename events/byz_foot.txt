# BYZANTINE NEWS

add_namespace = byzfoot

# Byzantine Football Championship 36
news_event = {
	id = byzfoot.1
	hidden = yes
	
	fire_only_once = yes
	
	trigger = {
		date > 1936.6.1
		date < 1936.9.1
		NOT = { has_global_flag = byz_foot_36 }
		country_exists = GRE
		GRE = { has_war = no }
	}

	mean_time_to_happen = {
		days = 14
	}
	
	immediate = {
		random_list = {
			20 = { GRE = { news_event = { id = byzfoot.2 } } }
			15 = { GRE = { news_event = { id = byzfoot.3 } } }
			15 = { GRE = { news_event = { id = byzfoot.4 } } }
			15 = { GRE = { news_event = { id = byzfoot.5 } } }
			15 = { GRE = { news_event = { id = byzfoot.6 } } }
			10 = { GRE = { news_event = { id = byzfoot.7 } } }
			10 = { GRE = { news_event = { id = byzfoot.8 } } }
		}
		set_global_flag = byz_foot_36
	}
	
	option = {
		name = byzfoot.36.a
	}
}
