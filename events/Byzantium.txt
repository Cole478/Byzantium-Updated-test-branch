### RECLAIMING THRACIA - BULGARIA

country_event = { 
	id = byzantium.1
	title = byzantium.1.t
	desc = byzantium.1.d
	picture = GFX_report_event_second_vienna_award_pact
	
	is_triggered_only = yes

	option = {
		name = byzantium.1.a # Claim
		ai_chance = {
			factor = 90
			modifier = {
				factor = 0
				any_country {
					OR = {
						owns_state = 48
						owns_state = 211
						owns_state = 212
						owns_state = 801
					}
					strength_ratio = {
						tag = GRE ratio > 1.5 # more than 150% the strength of Byzantium
					}
				}
			}
			modifier = {
				factor = 0
				any_other_country = {
					is_major = yes
					has_guaranteed = BUL
				}
			}
		}
		every_country {
			limit = {
				OR = {
					owns_state = 48
					owns_state = 211
					owns_state = 212
					owns_state = 801
				}
				AND = {
					NOT = {
						OR = {
							tag = GRE
							is_in_faction_with = GRE
							is_puppet_of = GRE
						}
					}
				}
			}
			country_event = { id = byzantium.2 days = 1 }
		}
	}
	
	option = {
		name = byzantium.1.b # Decline
		ai_chance = {
			factor = 10
		}
		news_event = { id = byznews.1 days = 1 }
	}
}
